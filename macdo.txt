<head>
  <link rel="stylesheet" href="http://cdn.jsdelivr.net/gh/highlightjs/cdn-release@9.12.0/build/styles/monokai-sublime.min.css">
  <link rel="stylesheet" href="style.css">
  <script src="http://cdn.jsdelivr.net/gh/highlightjs/cdn-release@9.12.0/build/highlight.min.js"></script>
  <script>
  const config = {
    language: 'html',
    lineBreak: true
  }
  </script>
</head>

<body>
  <input type="file" id="file">
  <pre><code class="code">
  </code></pre>
  <script>


  let speed = 50
  let i = 0
  let writing = false;
  function fileSelected() {
    document.querySelector('.code').classList.add(config.language)
    hljs.highlightBlock(document.querySelector(".code"));
    var selectedFile = document.getElementById('file').files[0];
    var reader = new FileReader();
    let text = ''
    reader.readAsText(selectedFile);
    reader.onload = function(event) {
      document.getElementById('file').classList.add('hidden')
      text = reader.result;
      console.log(text)
      arr = text.split('!ยง')
    };

    document.addEventListener('keydown', (event) => {
      if(event.keyCode == 32) {
      event.preventDefault();

      if(!writing) {
        typeWriter(arr)
        writing = true
    }
  }
  event.preventDefault();
  if(event.keyCode == 39 && speed > 5) speed-=5
  if(event.keyCode == 37) speed+=5
  })
}

function typeWriter(arr) {
  if (i < arr[0].length) {
    console.log(arr[0][i])
    if(!config.lineBreak) {
      if(arr[0][i] !== '\n') {
        document.querySelector(".code").append(arr[0][i]);
      }
    } else document.querySelector(".code").append(arr[0][i]);

    i++;
    if(arr[0][i] === '\n') {
    	hljs.highlightBlock(document.querySelector(".code"));
    	window.scrollTo(0,document.body.scrollHeight);
    }

    if(arr[0][i] === ' ') {
      typeWriter(arr)
      return null
    }
    else
    setTimeout(function () {
      typeWriter(arr)
    }, speed);
  }
  else {
    writing = false
    arr.shift()
    hljs.highlightBlock(document.querySelector(".code"));
    i = 0
  }
}
document.querySelector("#file").addEventListener('change',fileSelected)
  </script>
</body>
